\documentclass{beamer}
\usepackage[backend=bibtex]{biblatex}
\addbibresource{reference.bib}
\usepackage{tcolorbox}
\usepackage{enumitem}
\usepackage{varwidth}
\usepackage{adjustbox}

% Set the background color to black
\setbeamercolor{background canvas}{bg=black}

% Set the text color to white
\setbeamercolor{normal text}{fg=white}

% Define a custom box style
\tcbset{mybox/.style={
    colframe=black, % Box border color
    colback=white,  % Box background color
    left=0pt,       % Space on the left side of the box
    right=0pt,      % Space on the right side of the box
    top=0pt,        % No space above the box
    bottom=0pt,     % No space below the box
    boxsep=5pt,     % Padding inside the box
    sharp corners   % Sharp corners for the box
}}

% Adjust the box length based on the length of the item
\newcommand{\adjustedbox}[1]{%
  \begin{adjustbox}{minipage=\linewidth, center}
    \begin{tcolorbox}[mybox]
      #1
    \end{tcolorbox}%
  \end{adjustbox}%
}

\begin{document}

\begin{frame}
  \begin{enumerate}[label=\arabic*.]
    \item \adjustedbox{Short item}
    \item \adjustedbox{A longer item}
    \item \adjustedbox{An even longer item that spans multiple lines}
  \end{enumerate}
\end{frame}

\end{document}
